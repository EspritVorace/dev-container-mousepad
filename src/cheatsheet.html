⚛ Hooks React

const [v, setV] = useState(init)
setV(prev => prev + 1) — fonctionnelle (recommandé)
useEffect— effets : API, abonnements, DOM
useEffect(() => { ... return cleanup }, [deps])
useRef— ref mutable, pas de re-rendu
const ref = useRef<HTMLDivElement>(null)
useMemo— cache résultat calcul coûteux

useCallback— cache fonction entre rendus
useContext— lit le Provider parent le plus proche
useReducer— état complexe + dispatch actions
const [s, dispatch] = useReducer(fn, init)
useId— id unique SSR-safe
useTransition— update non-urgente (18+)
useDeferredValue— retarde valeur, priorise UI
useLayoutEffect— sync avant paint (mesures DOM)
